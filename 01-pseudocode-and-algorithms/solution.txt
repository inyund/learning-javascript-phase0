-- SOAL 2: PENILAIAN GRADE SISWA --
STORE "nama" WITHOUT ANY VALUE
STORE "nilai" WITHOUT ANY VALUE
STORE "grade" WITHOUT ANY VALUE

IF "nilai" IS BETWEEN 80 AND 100 THEN SET "grade" VALUE WITH "A"
ELSE IF "nilai" IS BETWEEN 65 AND 79 THEN SET "grade" VALUE WITH "B"
ELSE IF "nilai" IS BETWEEN 50 AND 64 THEN SET "grade" VALUE WITH "C"
ELSE IF "nilai" IS BETWEEN 35 AND 49 THEN SET "grade" VALUE WITH "D"
ELSE IF "nilai" IS BETWEEN 0 AND 34 THEN SET "grade" VALUE WITH "E"
ELSE SET "grade" VALUE WITH "Nilai Invalid"

DISPLAY "nama"
DISPLAY "grade"
